# Изменение политики запуска скриптов

#Restricted: Это наиболее строгий уровень. Запрещает запуск скриптов. 
#			Вы можете использовать PowerShell только для выполнения команд в командной строке. 
#			Это значение по умолчанию.

#AllSigned: Разрешает запуск только подписанных скриптов. 
#			Скрипты, подписанные надлежащим образом, могут выполняться.

#RemoteSigned: Разрешает запуск всех локальных скриптов, но для запуска скриптов, загруженных из интернета, 
#				требуется подпись.

#Unrestricted: Наименее строгий уровень. Позволяет запускать любые скрипты без каких-либо ограничений.

#Bypass: Полностью отключает проверку политики выполнения, что означает, 
#		что все скрипты могут выполняться без каких-либо ограничений.

#Undefined: Это означает, что политика выполнения не установлена. 
#			В этом случае PowerShell не применяет никаких ограничений к запуску скриптов.



#Установить политику выполнения для текущего пользователя, разрешив выполнение всех скриптов:
#Set-ExecutionPolicy Unrestricted -Scope CurrentUser

#Установить политику выполнения для всех пользователей на этом компьютере:
#Set-ExecutionPolicy RemoteSigned -Scope LocalMachine

#Установить политику выполнения для текущего пользователя, требующую подписи для скриптов из интернета:
#Set-ExecutionPolicy RemoteSigned -Scope CurrentUser

#Установить политику выполнения для текущего пользователя, разрешив выполнение только подписанных скриптов:
#Set-ExecutionPolicy AllSigned -Scope CurrentUser

#Отключить проверку политики выполнения:
#Set-ExecutionPolicy Bypass -Scope CurrentUser


#Временно изменить политику выполнения в рамках текущей сессии (не сохраняет настройки после перезапуска):
#Set-ExecutionPolicy RemoteSigned -Scope Process


#Обратите внимание, что для изменения политики выполнения PowerShell 
#вам может потребоваться запустить PowerShell с правами администратора.

Set-ExecutionPolicy Unrestricted -Scope LocalMachine




# Активация виртуальной среды
& "venv/Scripts/Activate.ps1"


# подключаю текущую директорию к переменной $env:PATH
$currentDirectory = Get-Location
$env:PATH += ";$currentDirectory"

# Погнали !
python start.py
